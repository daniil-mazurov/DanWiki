# Множественные условия
```go
a := 5
if a > 0 && a < 100 {
	fmt.Println("Positive")
} else if a == 0 {
	fmt.Println("Zero")
} else {
	fmt.Println("Negative")
}
```
# Инициализация переменной в условии
```go

if a := 100; a > 0 {
	fmt.Println("Positive")
}
```
# Switch
```go
a := 2
switch a {
case 1:
	fmt.Println("One")
case 2:
	fmt.Println("Two")
default:
	fmt.Println("Default")
}
```
# Switch с вызовом функции
```go

func mySum(x, y int) int {
	return x + y
}

func main() {
	switch res := mySum(1, 1); {
	case res > 0:
		fmt.Println("Positive")
	case res == 0:
		fmt.Println("Zero")
	default:
		fmt.Println("Negative")
	}
}
```
# Множественные условия в switch
```go
x := 2
switch x {
case 1, 2, 3:
	fmt.Println("Positive")
case -1, -2, -3:
	fmt.Println("Negative")
}
```
# Switch с fallthrough (падение в следующий блок кода)
```go
	x := 1
	switch x {
	case 1:
		fmt.Println("Positive”) // Выполнится
		fallthrough
	case 2:
		fmt.Println("Negative")  // Также выполнится!
	case 3:
		fmt.Println("Negative")
	}
```
